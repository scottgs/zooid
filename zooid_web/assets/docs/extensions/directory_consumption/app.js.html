function boot_dir(dirpath){
  fs.readdirSync("./"+dirpath+"/").forEach(function(file) {
    if( fs.lstatSync(dirpath+"/"+file).isDirectory() ){
      boot(dirpath+"/"+file,"app.js","");
    }
  });
}
